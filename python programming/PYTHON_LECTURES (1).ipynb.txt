        "\n",
        "class BankAccount:\n",
        "\n",
        "    # constructor\n",
        "    # private variable/protected variale/feature :self.__Feature\n",
        "    def __init__(self, owner, balance=0):\n",
        "        self.owner = owner            # public attribute/ feature\n",
        "        self.__balance = balance      # private attribute/ feature\n",
        "\n",
        "    # Getter method (read-only access to balance)\n",
        "    def get_balance(self):\n",
        "        return self.__balance\n",
        "\n",
        "    # Deposit method with validation\n",
        "    def deposit(self, amount):\n",
        "        if amount > 0:\n",
        "            self.__balance += amount  # amount + __balance\n",
        "            print(f\"Deposited: {amount}\")\n",
        "        else:\n",
        "            print(\"Invalid deposit amount!\")\n",
        "\n",
        "    # Withdraw method with validation\n",
        "    def withdraw(self, amount):\n",
